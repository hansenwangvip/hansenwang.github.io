tree.children.sort((a, b) => {  
  const INDEX_A = sortConfig.indexOf(a);  
  const INDEX_B = sortConfig.indexOf(b);  
  const HAS_A = INDEX_A > -1;  
  const HAS_B = INDEX_B > -1;  
  if (HAS_A && HAS_B) {  
    /* 对比二者顺序 */  
    return INDEX_A - INDEX_B;  
  }  
  if (HAS_A && !HAS_B) {  
    return 1;  
  }  
  if (!HAS_A && HAS_B) {  
    return -1;  
  }  
  return a.localeCompare(b);  
});
